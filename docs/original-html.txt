// ================================================================
// N8N NODE PARAMETER & SETTINGS RENDERER ENGINE

const N8nRenderer = (() => {
  function esc(str) {
    if (str === null || str === undefined) return '';
    const d = document.createElement('div');
    d.textContent = String(str);
    return d.innerHTML;
  }
  function isExpression(value) {
    return typeof value === 'string' && value.includes('{{') && value.includes('}}');
  }
  function humanizeKey(key) {
    return key.replace(/([A-Z])/g, ' $1').replace(/_/g, ' ').replace(/\b\w/g, c => c.toUpperCase()).trim();
  }
  function getNodeTypeName(type) {
    return (type || '').replace('n8n-nodes-base.', '').replace('@n8n/n8n-nodes-langchain.', '').replace('CUSTOM.', '');
  }
  function renderExpr(value, resolved) {
    let h = `<div class="n8n-expression-field"><div class="n8n-expression-header"><span class="n8n-expression-icon">/ₙ</span></div><div class="n8n-expression-body">${esc(value)}</div>`;
    if (resolved !== undefined && resolved !== null && resolved !== '') h += `<div class="n8n-expression-resolved">${esc(String(resolved))}</div>`;
    return h + '</div>';
  }
  function renderToggle(label, checked) {
    return `<div class="n8n-toggle-row"><span class="n8n-form-label">${esc(label)}</span><label class="n8n-toggle"><input type="checkbox" ${checked ? 'checked' : ''} disabled><span class="n8n-toggle-slider"></span></label></div>`;
  }
  function renderDropdown(label, value, options) {
    const opts = options || [value];
    let h = `<div class="n8n-form-group"><label class="n8n-form-label">${esc(label)}</label><select class="n8n-select" disabled>`;
    opts.forEach(o => { h += `<option ${o === value ? 'selected' : ''}>${esc(String(o))}</option>`; });
    return h + '</select></div>';
  }
  function renderCredField(label, name) {
    return `<div class="n8n-form-group"><label class="n8n-form-label">${esc(label)}</label><div class="n8n-credential-row"><select class="n8n-select" disabled><option>${esc(name)}</option></select><div class="n8n-credential-edit"><svg width="12" height="12" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><path d="M11 4H4a2 2 0 0 0-2 2v14a2 2 0 0 0 2 2h14a2 2 0 0 0 2-2v-7"/><path d="M18.5 2.5a2.121 2.121 0 0 1 3 3L12 15l-4 1 1-4 9.5-9.5z"/></svg></div></div></div>`;
  }
  function renderSplitSel(label, mode, value) {
    return `<div class="n8n-form-group"><label class="n8n-form-label">${esc(label)}</label><div class="n8n-split-select"><select class="n8n-select-small" disabled><option>${esc(mode || 'From list')}</option></select><select class="n8n-select" disabled><option>${esc(value || '')}</option></select></div></div>`;
  }
  function renderKVPair(n, v) {
    const nh = isExpression(n) ? renderExpr(n) : `<input class="n8n-input" value="${esc(n || '')}" disabled>`;
    const vh = isExpression(v) ? renderExpr(v) : `<input class="n8n-input" value="${esc(String(v || ''))}" disabled>`;
    return `<div class="n8n-kv-pair"><div class="n8n-form-group"><label class="n8n-form-sublabel">Name</label>${nh}</div><div class="n8n-form-group"><label class="n8n-form-sublabel">Value</label>${vh}</div></div>`;
  }
  function renderKVPairs(label, pairs) {
    if (!pairs || !Array.isArray(pairs) || !pairs.length) return '';
    let h = `<div class="n8n-form-group"><label class="n8n-form-label">${esc(label)}</label>`;
    pairs.forEach(p => { h += renderKVPair(p.name || p.key || '', p.value || ''); });
    return h + '</div>';
  }
  function renderValue(value, key) {
    if (value === null || value === undefined) return '';
    if (typeof value === 'boolean') return renderToggle(humanizeKey(key), value);
    if (typeof value === 'string') {
      if (isExpression(value)) return renderExpr(value);
      if (value.includes('\n') && value.length > 60) return `<textarea class="n8n-textarea" disabled>${esc(value)}</textarea>`;
      return `<input class="n8n-input" type="text" value="${esc(value)}" disabled>`;
    }
    if (typeof value === 'number') return `<input class="n8n-input" type="text" value="${value}" disabled>`;
    if (Array.isArray(value)) {
      if (!value.length) return `<div class="n8n-no-properties">No items</div>`;
      return value.map((item, i) => {
        if (typeof item === 'object' && item !== null) return `<div class="n8n-nested-block">${renderParamObj(item)}</div>`;
        return `<div class="n8n-form-group"><label class="n8n-form-label">[${i}]</label>${renderValue(item, String(i))}</div>`;
      }).join('');
    }
    if (typeof value === 'object') return renderParamObj(value);
    return `<input class="n8n-input" type="text" value="${esc(String(value))}" disabled>`;
  }
  const SKIP = new Set(['__rl','matchingColumns','schema','mappingMode','convertFieldsToString','cachedResultUrl','cachedResultName','typeVersion','id','webhookId']);
  function renderParamObj(obj, depth) {
    if (!obj || typeof obj !== 'object') return '';
    let h = '';
    Object.entries(obj).forEach(([key, value]) => {
      if (SKIP.has(key) || value === undefined) return;
      if (typeof value === 'object' && value !== null && value.__rl) {
        const m = value.mode === 'list' ? 'From list' : (value.mode === 'id' ? 'By ID' : value.mode);
        h += renderSplitSel(humanizeKey(key), m, value.cachedResultName || value.value || '');
        return;
      }
      if (key === 'value' && typeof value === 'object' && value !== null && !Array.isArray(value)) {
        Object.entries(value).forEach(([vk, vv]) => {
          h += `<div class="n8n-workflow-input"><div class="n8n-workflow-input-name"><span class="icon">🔒</span> ${esc(vk)}</div>${renderValue(vv, vk)}</div>`;
        });
        return;
      }
      if (typeof value === 'boolean') { h += renderToggle(humanizeKey(key), value); return; }
      if (key === 'conditions' && typeof value === 'object' && value.conditions) { h += renderConditions(value); return; }
      if (key === 'messages' && typeof value === 'object') { h += renderMessages(value); return; }
      if ((key === 'headerParameters' || key === 'queryParameters') && typeof value === 'object') {
        const p = value.parameters || value;
        if (Array.isArray(p)) h += renderKVPairs(humanizeKey(key), p);
        return;
      }
      if (key === 'options' && typeof value === 'object') {
        const entries = Object.entries(value).filter(([k]) => !SKIP.has(k));
        h += `<div class="n8n-options-section"><div class="n8n-options-header">Options</div>`;
        if (!entries.length) h += `<div class="n8n-no-properties">No properties</div>`;
        else entries.forEach(([ok, ov]) => { h += `<div class="n8n-form-group"><label class="n8n-form-sublabel">${humanizeKey(ok)}</label>${renderValue(ov, ok)}</div>`; });
        h += `<div class="n8n-add-btn">Add option</div></div>`;
        return;
      }
      if (typeof value === 'object' && value !== null && !Array.isArray(value)) {
        const entries = Object.entries(value).filter(([k]) => !SKIP.has(k));
        if (!entries.length) return;
        const isSimple = entries.every(([, v]) => typeof v !== 'object' || v === null);
        if (isSimple && entries.length <= 3) { entries.forEach(([ok, ov]) => { h += `<div class="n8n-form-group"><label class="n8n-form-label">${humanizeKey(ok)}</label>${renderValue(ov, ok)}</div>`; }); return; }
        h += `<div class="n8n-section-header">${humanizeKey(key)}</div>`;
        h += renderParamObj(value);
        return;
      }
      if (Array.isArray(value)) {
        if (!value.length) return;
        h += `<div class="n8n-form-group"><label class="n8n-form-label">${humanizeKey(key)}</label>`;
        if (value[0] && typeof value[0] === 'object' && ('name' in value[0] || 'key' in value[0]) && 'value' in value[0]) value.forEach(item => { h += renderKVPair(item.name || item.key || '', item.value || ''); });
        else h += renderValue(value, key);
        h += '</div>';
        return;
      }
      h += `<div class="n8n-form-group"><label class="n8n-form-label">${humanizeKey(key)}</label>${renderValue(value, key)}</div>`;
    });
    return h;
  }
  function renderConditions(co) {
    const conds = co.conditions || [];
    let h = `<div class="n8n-form-group"><label class="n8n-form-label">Conditions</label>`;
    conds.forEach(c => {
      const lv = c.leftValue || '', rv = c.rightValue || '', op = c.operator || {};
      const opMap = {equals:'equals',notEquals:'not equals',contains:'contains',empty:'is empty',notEmpty:'is not empty',true:'is true',false:'is false',gt:'>',gte:'>=',lt:'<',lte:'<='};
      const opLabel = opMap[op.operation] || op.operation || '=';
      h += '<div class="n8n-condition-row">';
      h += isExpression(lv) ? `<div class="n8n-expression-field" style="flex:1"><div class="n8n-expression-header"><span class="n8n-expression-icon">/ₙ</span></div><div class="n8n-expression-body">${esc(lv)}</div></div>` : `<input class="n8n-input" style="flex:1" value="${esc(String(lv))}" disabled>`;
      h += `<div class="n8n-condition-operator">${esc(opLabel)}</div>`;
      if (!op.singleValue) {
        h += isExpression(rv) ? `<div class="n8n-expression-field" style="flex:1"><div class="n8n-expression-header"><span class="n8n-expression-icon">/ₙ</span></div><div class="n8n-expression-body">${esc(rv)}</div></div>` : `<input class="n8n-input" style="flex:1" value="${esc(String(rv))}" disabled>`;
      }
      h += '</div>';
    });
    h += `<div class="n8n-add-btn">Add condition</div></div>`;
    return h;
  }
  function renderMessages(mo) {
    let arr = mo && mo.values ? mo.values : (Array.isArray(mo) ? mo : null);
    if (!arr) return '';
    let h = '<div class="n8n-section-header">Messages</div>';
    arr.forEach(msg => {
      h += '<div class="n8n-message-block">';
      const content = msg.content || msg.message || '';
      if (content) h += `<div class="n8n-form-group"><label class="n8n-form-label">Prompt</label>${isExpression(content) ? renderExpr(content) : `<textarea class="n8n-textarea" disabled>${esc(content)}</textarea>`}</div>`;
      if (msg.role !== undefined) h += renderDropdown('Role', msg.role, ['System', 'User', 'Assistant', msg.role]);
      h += '</div>';
    });
    h += '<div class="n8n-add-btn">Add Message</div>';
    return h;
  }
  function humanizeAuth(a) { return {predefinedCredentialType:'Predefined Credential Type',genericCredentialType:'Generic Credential Type',httpBasicAuth:'Basic Auth',httpHeaderAuth:'Header Auth',httpBearerAuth:'Bearer Auth',none:'None'}[a] || a; }
  function renderOptionsSection(opts) {

    let h = '<div class="n8n-options-section"><div class="n8n-options-header">Options</div>';
    if (!entries.length) h += '<div class="n8n-no-properties">No properties</div>';
    else entries.forEach(([k, v]) => { h += `<div class="n8n-form-group"><label class="n8n-form-sublabel">${humanizeKey(k)}</label>${renderValue(v, k)}</div>`; });

    return h;
  }

      if (p.headerParameters?.parameters) h += renderKVPairs('Header Parameters', p.headerParameters.parameters);
    }
    if (p.sendBody !== undefined) h += renderToggle('Send Body', p.sendBody);
    if (p.sendBody) {
      if (p.contentType) h += renderDropdown('Body Content Type', p.contentType === 'json' ? 'JSON' : p.contentType);
      if (p.specifyBody) h += renderDropdown('Specify Body', p.specifyBody === 'json' ? 'Using JSON' : p.specifyBody);

document.addEventListener('DOMContentLoaded', () => {
  initSearchForm();
  initCanvas();
  
  const params = new URLSearchParams(window.location.search);
  const wfId = params.get('workflow_id');
  const execId = params.get('execution_id');
  
  if (wfId) {
    document.getElementById('workflowIdInput').value = wfId;
    updateSearchButton();
  }
  if (execId) {
    document.getElementById('executionIdInput').value = execId;
    updateSearchButton();
    handleSearch();
  }
});

function initSearchForm() {
  const wfInput = document.getElementById('workflowIdInput');
  const execInput = document.getElementById('executionIdInput');
  
  wfInput.addEventListener('input', updateSearchButton);
  execInput.addEventListener('input', updateSearchButton);
  
  [wfInput, execInput].forEach(input => {
    input.addEventListener('keydown', (e) => {
      if (e.key === 'Enter' && !document.getElementById('searchBtn').disabled) {
        handleSearch();
      }
    });
  });
}

function updateSearchButton() {
  const wfId = document.getElementById('workflowIdInput').value.trim();
  const execId = document.getElementById('executionIdInput').value.trim();
  document.getElementById('searchBtn').disabled = !wfId && !execId;
}

function initCanvas() {
  const container = document.getElementById('canvasContainer');
  
  container.addEventListener('mousedown', (e) => {
    if (e.target.closest('.node') || container.classList.contains('empty')) return;
    state.isPanning = true;
    state.startPanX = e.clientX - state.panX;
    state.startPanY = e.clientY - state.panY;
    container.style.cursor = 'grabbing';
  });
  
  document.addEventListener('mousemove', (e) => {
    if (state.isPanning) {
      state.panX = e.clientX - state.startPanX;
      state.panY = e.clientY - state.startPanY;
      updateCanvasTransform();
    }
  });
  
  document.addEventListener('mouseup', () => {
    state.isPanning = false;
    const container = document.getElementById('canvasContainer');
    if (!container.classList.contains('empty')) {
      container.style.cursor = 'grab';
    }
  });
  
  container.addEventListener('wheel', (e) => {
    if (container.classList.contains('empty')) return;
    e.preventDefault();
    
    if (e.ctrlKey) {
      const rect = container.getBoundingClientRect();
      const mx = e.clientX - rect.left;
      const my = e.clientY - rect.top;
      const factor = e.deltaY > 0 ? 0.9 : 1.1;
      const newZoom = Math.max(0.1, Math.min(2.5, state.zoom * factor));
      const ratio = newZoom / state.zoom;
      state.panX = mx - (mx - state.panX) * ratio;
      state.panY = my - (my - state.panY) * ratio;
      state.zoom = newZoom;
    } else if (e.shiftKey) {
      state.panX -= e.deltaY;
    } else {
      state.panY -= e.deltaY;
    }
    
    updateCanvasTransform();
  }, { passive: false });
}

function updateCanvasTransform() {
  document.getElementById('canvas').style.transform = 
    `translate(${state.panX}px, ${state.panY}px) scale(${state.zoom})`;
}

// ==================== API CALLS ====================
async function handleSearch() {
  const wfId = document.getElementById('workflowIdInput').value.trim();
  const execId = document.getElementById('executionIdInput').value.trim();
  
  if (!wfId && !execId) return;
  
  state.workflowId = wfId || null;
  state.offset = 0;
  state.hasMore = true;
  
  showExecListLoading();
  
  try {
    if (execId) {
      await loadExecutionDetails(execId);
    } else {
      await loadExecutionList();
    }
  } catch (error) {
    console.error('Search error:', error);
    showExecListError(error.message);
  }
}

async function loadExecutionList() {
  const params = new URLSearchParams({
    workflow_id: state.workflowId || '',
    offset: state.offset,
    limit: CONFIG.PAGE_SIZE
  });
  
  const response = await fetch(`${CONFIG.API_BASE}${CONFIG.ENDPOINTS.LIST}?${params}`);
  const data = await response.json();
  
  if (state.offset === 0) {
    state.executions = data.executions;
    state.workflowName = data.workflowName;
    updateWorkflowName(data.workflowName);
  } else {
    state.executions = [...state.executions, ...data.executions];
  }
  
  state.hasMore = data.executions.length === CONFIG.PAGE_SIZE;
  renderExecutionList();
}

async function loadMoreExecutions() {
  if (state.isLoading || !state.hasMore) return;
  
  state.isLoading = true;
  state.offset += CONFIG.PAGE_SIZE;
  
  try {
    await loadExecutionList();
  } catch (error) {
    console.error('Load more error:', error);
  } finally {
    state.isLoading = false;
  }
}

async function loadExecutionDetails(execId) {
  showCanvasLoading();
  
  const response = await fetch(`${CONFIG.API_BASE}${CONFIG.ENDPOINTS.DETAILS}?execution_id=${execId}`);
  const data = await response.json();
  
  if (!data) {
    showExecListError('Execution not found');
    return;
  }
  
  state.currentExecId = execId;
  state.currentExecData = data;
  state.workflow = data.workflow;
  state.workflowName = data.workflow?.name || data.workflowName;
  
  updateWorkflowName(state.workflowName);
  
  if (state.executions.length === 0) {
    state.executions = [{
      id: execId,
      status: data.status,
      startedAt: data.startedAt,
      stoppedAt: data.stoppedAt,
      workflowName: state.workflowName
    }];
    renderExecutionList();
  }
  
  processExecutionData(data);
  renderCanvas();
  updateInfoBar(data);
  
  selectExecutionInList(execId);
}

function selectExecution(execId) {
  if (state.currentExecId === execId) return;
  loadExecutionDetails(execId);
}

// NOTE: workflow.nodes comes from execution.workflowData (snapshot at execution time),
// so there are no "deleted" nodes — every node in the snapshot existed when this ran.
function processExecutionData(data) {
  const workflow = data.workflow || {};
  const runData = data.execution?.data?.resultData?.runData || {};
  const workflowNodes = workflow.nodes || [];
  const workflowConnections = workflow.connections || {};

  // Reset port counts
  state.nodeInputCounts = {};
  state.nodeOutputCounts = {};

  // STEP 1: Calculate INPUT port counts from connections
  Object.entries(workflowConnections).forEach(([fromName, outputs]) => {
    if (!outputs.main) return;
    outputs.main.forEach((targets) => {
      if (!targets) return;
      targets.forEach(target => {
        const targetIndex = target.index || 0;
        state.nodeInputCounts[target.node] = Math.max(
          state.nodeInputCounts[target.node] || 0,
          targetIndex + 1
        );
      });
    });
  });

  // STEP 2: Calculate OUTPUT port counts from runData + connections
  const triggerTypes = [
    'n8n-nodes-base.webhook',
    'n8n-nodes-base.manualTrigger',
    'n8n-nodes-base.scheduleTrigger',
    'n8n-nodes-base.executeWorkflowTrigger'
  ];

  workflowNodes.forEach(node => {
    // Trigger nodes have no inputs
    if (triggerTypes.includes(node.type)) {
      state.nodeInputCounts[node.name] = 0;
    }

    const nodeRuns = runData[node.name];
    let outputCount = 1;

    if (nodeRuns && nodeRuns.length > 0) {
      const lastRun = nodeRuns[nodeRuns.length - 1];
      if (lastRun.data && lastRun.data.main) {
        outputCount = lastRun.data.main.length;
      }
    }

    if (node.onError === 'continueErrorOutput') {
      outputCount = Math.max(outputCount, 2);
    }

    const nodeConnections = workflowConnections[node.name];
    if (nodeConnections && nodeConnections.main) {
      outputCount = Math.max(outputCount, nodeConnections.main.length);
    }

    state.nodeOutputCounts[node.name] = outputCount;
  });

  // STEP 3: Process nodes — all sourced from the execution snapshot
  state.nodes = workflowNodes.map(node => {
    const nodeRuns = runData[node.name] || [];
    const lastRun = nodeRuns[nodeRuns.length - 1];
    const hasRun = !!lastRun;
    const hasError = !!(lastRun?.error);

    let status = 'not-executed';
    if (hasError) status = 'error';
    else if (hasRun) status = 'success';

    const outputData = lastRun?.data?.main?.map(item => item.map(i => i.json || {})) || [];
    const outputCount = outputData.reduce((sum, arr) => sum + (arr?.length || 0), 0);
    const inputSources = lastRun?.source || [];

    return {
      id: node.id,
      name: node.name,
      type: node.type,
      typeVersion: node.typeVersion,
      position: node.position || [0, 0],
      parameters: node.parameters || {},
      credentials: node.credentials || null,
      settings: node.settings || {},
      onError: node.onError,
      alwaysOutputData: node.alwaysOutputData || false,
      executeOnce: node.executeOnce || false,
      retryOnFail: node.retryOnFail || false,
      maxTries: node.maxTries || 3,
      waitBetweenTries: node.waitBetweenTries || 1000,
      notes: node.notes || '',
      notesInFlow: node.notesInFlow || false,
      status,
      hasRun,
      hasError,
      error: lastRun?.error || null,
      startTime: lastRun?.startTime || null,
      executionTime: lastRun?.executionTime || null,
      inputSources,
      outputData,
      outputCount,
      runData: lastRun,
      allRuns: nodeRuns,
      inputPortCount: state.nodeInputCounts[node.name] || 0,
      outputPortCount: state.nodeOutputCounts[node.name] || 1,
      isDeleted: false
    };
  });

  // STEP 4: Process connections
  state.connections = [];
  Object.entries(workflowConnections).forEach(([fromName, outputs]) => {
    if (!outputs.main) return;
    outputs.main.forEach((targets, outputIndex) => {
      if (!targets) return;
      targets.forEach(target => {
        const fromNode = state.nodes.find(n => n.name === fromName);
        const toNode = state.nodes.find(n => n.name === target.node);
        if (fromNode && toNode) {
          // Aggregate item count across ALL runs for this output port
          let itemCount = 0;
          const nodeRuns = runData[fromName] || [];
          nodeRuns.forEach(run => {
            const portData = run?.data?.main?.[outputIndex];
            if (portData) {
              itemCount += portData.length;
            }
          });

          state.connections.push({
            from: fromNode.id,
            fromName: fromName,
            to: toNode.id,
            toName: target.node,
            fromPort: outputIndex,
            toPort: target.index || 0,
            itemCount: itemCount,
            isDeletedConnection: false
          });
        }
      });
    });
  });
}

/**
 * Calculate the dynamic height of a node based on port count
 * @param {number} inputCount - Number of input ports
 * @param {number} outputCount - Number of output ports
 * @returns {number} - Node height in pixels
 */
function getNodeHeight(inputCount, outputCount) {
  const baseNodeHeight = 75;
  const minPortSpacing = 25; // Minimum spacing between ports
  const maxPorts = Math.max(inputCount || 0, outputCount || 1);
  
  if (maxPorts <= 2) {
    return baseNodeHeight;
  }
  
  // Calculate required height for ports with minimum spacing
  const requiredHeight = (maxPorts + 1) * minPortSpacing;
  return Math.max(baseNodeHeight, requiredHeight);
}

/**
 * Calculate the Y position for a specific port on a node
 * Uses formula: y = (nodeHeight / (portCount + 1)) * (index + 1)
 * This distributes ports evenly: 1 port at 50%, 2 ports at 33%/66%, etc.
 * @param {number} portIndex - The index of the port (0-based)
 * @param {number} totalPorts - Total number of ports
 * @param {number} nodeHeight - Height of the node box
 * @returns {number} - Y offset from top of node
 */
function getPortYPosition(portIndex, totalPorts, nodeHeight) {
  if (totalPorts <= 0) {
    return nodeHeight / 2;
  }
  
  // Formula: y = (nodeHeight / (portCount + 1)) * (index + 1)
  // This creates N+1 equal segments and places ports at segment boundaries
  return (nodeHeight / (totalPorts + 1)) * (portIndex + 1);
}

/**
 * Generate connection path - uses Bezier for forward, Orthogonal for backward
 */
function generateConnectionPath(sourceX, sourceY, targetX, targetY) {
  const dx = targetX - sourceX;
  
  if (dx >= 40) {
    // Forward connection: use Bezier curve
    return generateBezierPath(sourceX, sourceY, targetX, targetY);
  } else {
    // Backward connection: use Orthogonal/Manhattan path
    return generateOrthogonalPath(sourceX, sourceY, targetX, targetY);
  }
}

/**
 * Generate a Bezier curve path for forward connections
 * Control points offset by Math.abs(targetX - sourceX) / 2
 */
function generateBezierPath(sourceX, sourceY, targetX, targetY) {
  const dx = targetX - sourceX;
  const controlOffset = Math.max(Math.abs(dx) / 2, 40);
  
  const cx1 = sourceX + controlOffset;
  const cy1 = sourceY;
  const cx2 = targetX - controlOffset;
  const cy2 = targetY;
  
  return `M ${sourceX} ${sourceY} C ${cx1} ${cy1}, ${cx2} ${cy2}, ${targetX} ${targetY}`;
}

/**
 * Generate an Orthogonal/Manhattan path for backward connections
 * Path: Right -> Down -> Left -> Up -> Right (with rounded corners)
 */
function generateOrthogonalPath(sourceX, sourceY, targetX, targetY) {
  const r = 8; // Corner radius
  const horizontalGap = 30; // How far to extend right/left before turning
  const verticalPadding = 50; // How far below/above to route
  
  // Determine if we go below or above based on positions
  const goBelow = true; // For backward connections, typically go below
  const bottomY = Math.max(sourceY, targetY) + verticalPadding;
  
  const rightX = sourceX + horizontalGap;
  const leftX = targetX - horizontalGap;
  
  // Build orthogonal path with quadratic curves for rounded corners
  let path = `M ${sourceX} ${sourceY}`;
  
  // 1. Go right from source
  path += ` H ${rightX - r}`;
  
  // 2. Round corner: turn down
  path += ` Q ${rightX} ${sourceY}, ${rightX} ${sourceY + r}`;
  
  // 3. Go down to bottom level
  path += ` V ${bottomY - r}`;
  
  // 4. Round corner: turn left
  path += ` Q ${rightX} ${bottomY}, ${rightX - r} ${bottomY}`;
  
  // 5. Go left to target column
  path += ` H ${leftX + r}`;
  
  // 6. Round corner: turn up
  path += ` Q ${leftX} ${bottomY}, ${leftX} ${bottomY - r}`;
  
  // 7. Go up to target level
  path += ` V ${targetY + r}`;
  
  // 8. Round corner: turn right into target
  path += ` Q ${leftX} ${targetY}, ${leftX + r} ${targetY}`;
  
  // 9. Go right to target
  path += ` H ${targetX}`;
  
  return path;
}

/**
 * Calculate the midpoint of a connection for badge placement
 */
function getConnectionMidpoint(sourceX, sourceY, targetX, targetY) {
  const dx = targetX - sourceX;
  
  if (dx >= 40) {
    // Forward: midpoint of Bezier curve
    const controlOffset = Math.max(Math.abs(dx) / 2, 40);
    const cx1 = sourceX + controlOffset;
    const cy1 = sourceY;
    const cx2 = targetX - controlOffset;
    const cy2 = targetY;
    
    const t = 0.5;
    const mt = 1 - t;
    const x = mt*mt*mt*sourceX + 3*mt*mt*t*cx1 + 3*mt*t*t*cx2 + t*t*t*targetX;
    const y = mt*mt*mt*sourceY + 3*mt*mt*t*cy1 + 3*mt*t*t*cy2 + t*t*t*targetY;
    
    return { x, y };
  } else {
    // Backward: midpoint is on the bottom horizontal segment
    const bottomY = Math.max(sourceY, targetY) + 50;
    const midX = (sourceX + targetX) / 2;
    return { x: midX, y: bottomY };
  }
}

function zoomToFit() {
  if (state.nodes.length === 0) return;
  
  const container = document.getElementById('canvasContainer');
  const rect = container.getBoundingClientRect();
  const nodeWidth = 75;
  
  let minX = Infinity, minY = Infinity, maxX = -Infinity, maxY = -Infinity;
  state.nodes.forEach(n => {
    const inputCount = n.inputPortCount || 0;
    const outputCount = n.outputPortCount || 1;
    const nodeHeight = getNodeHeight(inputCount, outputCount);
    
    minX = Math.min(minX, n.position[0]);
    minY = Math.min(minY, n.position[1]);
    maxX = Math.max(maxX, n.position[0] + nodeWidth);
    maxY = Math.max(maxY, n.position[1] + nodeHeight + 35); // +35 for labels
  });
  
  const padding = 80;
  const contentWidth = maxX - minX + padding * 2;
  const contentHeight = maxY - minY + padding * 2;
  
  const scaleX = rect.width / contentWidth;
  const scaleY = rect.height / contentHeight;
  state.zoom = Math.min(scaleX, scaleY, 1);
  
  state.panX = (rect.width - contentWidth * state.zoom) / 2 - minX * state.zoom + padding * state.zoom;
  state.panY = (rect.height - contentHeight * state.zoom) / 2 - minY * state.zoom + padding * state.zoom;
  
  updateCanvasTransform();
}

function updateInputSourceSelect(node) {
  const trigger = document.getElementById('inputNodeTrigger');
  const list = document.getElementById('inputNodeList');

  // Find direct parents for default selection
  const parentConnections = state.connections.filter(c => c.to === node.id);
  const directParentIds = new Set(parentConnections.map(c => c.from));

  // Get the current node's earliest startTime
  const currentEarliestStart = (node.allRuns || []).reduce((min, r) => {
    return r.startTime && (min === null || r.startTime < min) ? r.startTime : min;
  }, null);

  // Collect all previously executed nodes
  let candidates = state.nodes.filter(n => {
    if (!n.hasRun || n.id === node.id) return false;
    if (currentEarliestStart === null) return true;
    const nEarliestStart = (n.allRuns || []).reduce((min, r) => {
      return r.startTime && (min === null || r.startTime < min) ? r.startTime : min;
    }, null);
    return nEarliestStart !== null && nEarliestStart <= currentEarliestStart;
  });

  // Sort by earliest startTime ascending
  candidates.sort((a, b) => {
    const aStart = (a.allRuns || []).reduce((min, r) => r.startTime && (min === null || r.startTime < min) ? r.startTime : min, null);
    const bStart = (b.allRuns || []).reduce((min, r) => r.startTime && (min === null || r.startTime < min) ? r.startTime : min, null);
    return (aStart || 0) - (bStart || 0);
  });

  if (candidates.length === 0) {
    document.getElementById('inputNodeIcon').innerHTML = '';
    document.getElementById('inputNodeLabel').textContent = 'No input data';
    list.innerHTML = '';
    state._inputSourceNodeId = '';
    return;
  }

  // Default to first direct parent, or first candidate
  const defaultNode = candidates.find(n => directParentIds.has(n.id)) || candidates[0];
  state._inputSourceNodeId = defaultNode.id;

  // Render trigger
  document.getElementById('inputNodeIcon').innerHTML = makeIconHtml(defaultNode.type);
  document.getElementById('inputNodeLabel').textContent = defaultNode.name;

  // Render list
  list.innerHTML = candidates.map(n => {
    const selected = n.id === defaultNode.id ? ' selected' : '';
    return `<div class="icon-dropdown-option${selected}" data-node-id="${n.id}" onclick="selectInputNode('${n.id}')">
      ${makeIconHtml(n.type)}
      <span style="overflow:hidden;text-overflow:ellipsis;white-space:nowrap">${escapeHtml(n.name)}</span>
    </div>`;
  }).join('');
}

// ==================== UTILITIES ====================
function calcDuration(start, end) {
  const ms = end - start;
  if (ms < 1000) return `${ms}ms`;
  if (ms < 60000) return `${(ms / 1000).toFixed(2)}s`;
  const mins = Math.floor(ms / 60000);
  const secs = Math.floor((ms % 60000) / 1000);
  return `${mins}m ${secs}s`;
}

function escapeHtml(str) {
  if (str === null || str === undefined) return '';
  const div = document.createElement('div');
  div.textContent = String(str);
  return div.innerHTML;
}

function syntaxHighlight(json) {
  if (!json) return '';
  return json
    .replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;')
    .replace(/("(\\u[a-zA-Z0-9]{4}|\\[^u]|[^\\"])*"(\s*:)?|\b(true|false|null)\b|-?\d+(?:\.\d*)?(?:[eE][+\-]?\d+)?)/g, (match) => {
      let cls = 'json-number';
      if (/^"/.test(match)) {
        cls = /:$/.test(match) ? 'json-key' : 'json-string';
      } else if (/true|false/.test(match)) {
        cls = 'json-boolean';
      } else if (/null/.test(match)) {
        cls = 'json-null';
      }
      return `<span class="${cls}">${match}</span>`;
    });
}

// ==================== COLLAPSIBLE JSON (Feature 7) ====================
function renderCollapsibleJson(data, indent) {
  if (indent === undefined) indent = 0;
  const pad = '  '.repeat(indent);
  const padInner = '  '.repeat(indent + 1);
  
  if (data === null) return `<span class="json-null">null</span>`;
  if (typeof data === 'boolean') return `<span class="json-boolean">${data}</span>`;
  if (typeof data === 'number') return `<span class="json-number">${data}</span>`;
  if (typeof data === 'string') return `<span class="json-string">"${escapeHtml(data)}"</span>`;
  
  if (Array.isArray(data)) {
    if (data.length === 0) return `[]`;
    const uid = 'jc_' + Math.random().toString(36).substr(2, 9);
    const count = data.length;
    const items = data.map((item, i) => {
      const comma = i < data.length - 1 ? ',' : '';
      return `${padInner}${renderCollapsibleJson(item, indent + 1)}${comma}`;
    }).join('\n');
    
    return `<span class="json-toggle" data-uid="${uid}" onclick="toggleJsonBlock(event, '${uid}')">[</span><span id="${uid}" class="json-block">\n${items}\n${pad}</span><span class="json-toggle" data-uid="${uid}" onclick="toggleJsonBlock(event, '${uid}')">]</span><span id="${uid}_collapsed" class="json-collapsed-indicator" style="display:none" onclick="toggleJsonBlock(event, '${uid}')">[...] // ${count} item${count !== 1 ? 's' : ''}</span>`;
  }
  
  if (typeof data === 'object') {
    const keys = Object.keys(data);
    if (keys.length === 0) return `{}`;
    const uid = 'jc_' + Math.random().toString(36).substr(2, 9);
    const count = keys.length;
    const items = keys.map((key, i) => {
      const comma = i < keys.length - 1 ? ',' : '';
      return `${padInner}<span class="json-key">"${escapeHtml(key)}"</span>: ${renderCollapsibleJson(data[key], indent + 1)}${comma}`;
    }).join('\n');
    
    return `<span class="json-toggle" data-uid="${uid}" onclick="toggleJsonBlock(event, '${uid}')">{</span><span id="${uid}" class="json-block">\n${items}\n${pad}</span><span class="json-toggle" data-uid="${uid}" onclick="toggleJsonBlock(event, '${uid}')">}</span><span id="${uid}_collapsed" class="json-collapsed-indicator" style="display:none" onclick="toggleJsonBlock(event, '${uid}')">{...} // ${count} item${count !== 1 ? 's' : ''}</span>`;
  }
  
  return escapeHtml(String(data));
}

function toggleJsonBlock(event, uid) {
  event.stopPropagation();
  const block = document.getElementById(uid);
  const collapsed = document.getElementById(uid + '_collapsed');
  // Find all toggles for this uid
  const toggles = document.querySelectorAll(`.json-toggle[data-uid="${uid}"]`);
  
  if (!block || !collapsed) return;
  
  if (block.style.display === 'none') {
    // Expand
    block.style.display = '';
    collapsed.style.display = 'none';
    toggles.forEach(t => t.style.display = '');
  } else {
    // Collapse
    block.style.display = 'none';
    collapsed.style.display = '';
    // Hide the closing bracket toggle
    if (toggles.length > 1) toggles[1].style.display = 'none';
  }
}

function togglePanelSearch(panel) {
  const field = document.getElementById(`${panel}SearchField`);
  if (field.classList.contains('expanded')) {
    field.classList.remove('expanded');
    field.value = '';
    handlePanelSearch(panel); // clear search
  } else {
    field.classList.add('expanded');
    setTimeout(() => field.focus(), 50);
  }
}

function handlePanelSearch(panel) {
  const query = document.getElementById(`${panel}SearchField`).value;
  const targetId = panel === 'input' ? 'inputJson' : 'outputContent';
  const el = document.getElementById(targetId);
  if (!el) return;
  
  // Find the json-viewer inside (output might wrap it)
  const viewer = el.classList.contains('json-viewer') ? el : el.querySelector('.json-viewer');
  if (!viewer) return;
  
  // Remove old highlights
  viewer.querySelectorAll('.search-match-line').forEach(line => {
    line.classList.remove('search-match-line');
  });
  viewer.querySelectorAll('.search-match-text').forEach(span => {
    span.outerHTML = span.textContent;
  });
  
  if (!query || query.length === 0) return;
  
  // Search through the text content line by line
  const lines = viewer.innerHTML.split('\n');
  const lowerQuery = query.toLowerCase();
  
  const highlighted = lines.map(line => {
    const plainText = line.replace(/<[^>]*>/g, '');
    if (plainText.toLowerCase().includes(lowerQuery)) {
      // Highlight matching text - we need to be careful with HTML tags
      let result = line;
      // Simple approach: wrap entire line and highlight text matches
      const regex = new RegExp(`(${query.replace(/[.*+?^${}()|[\]\\]/g, '\\$&')})`, 'gi');
      // Highlight within text nodes only (between tags)
      result = result.replace(/>([^<]*)</g, (match, textContent) => {
        const highlighted = textContent.replace(regex, '<span class="search-match-text">$1</span>');
        return `>${highlighted}<`;
      });
      return `<span class="search-match-line">${result}</span>`;
    }
    return line;
  });
  
  viewer.innerHTML = highlighted.join('\n');
}

// ==================== RESIZABLE CENTER PANEL (Feature 6) ====================
function initResizeHandles() {
  const leftHandle = document.getElementById('resizeHandleLeft');
  const rightHandle = document.getElementById('resizeHandleRight');
  const centerPanel = document.getElementById('centerPanelWrap');
  
  if (!leftHandle || !rightHandle || !centerPanel) return;
  
  let isResizing = false;
  let startX = 0;
  let startWidth = 0;
  let side = '';
  
  function startResize(e, resizeSide) {
    e.preventDefault();
    e.stopPropagation();
    isResizing = true;
    startX = e.clientX;
    startWidth = centerPanel.offsetWidth;
    side = resizeSide;
    document.body.style.cursor = 'col-resize';
    document.body.style.userSelect = 'none';
    if (side === 'left') leftHandle.classList.add('active');
    else rightHandle.classList.add('active');
  }
  
  leftHandle.addEventListener('mousedown', (e) => startResize(e, 'left'));
  rightHandle.addEventListener('mousedown', (e) => startResize(e, 'right'));
  
  document.addEventListener('mousemove', (e) => {
    if (!isResizing) return;
    const delta = side === 'left' ? (startX - e.clientX) : (e.clientX - startX);
    const newWidth = Math.max(250, Math.min(700, startWidth + delta));
    centerPanel.style.width = newWidth + 'px';
  });
  
  document.addEventListener('mouseup', () => {
    if (isResizing) {
      isResizing = false;
      document.body.style.cursor = '';
      document.body.style.userSelect = '';
      leftHandle.classList.remove('active');
      rightHandle.classList.remove('active');
    }
  });
}

// Init resize on DOM ready
document.addEventListener('DOMContentLoaded', () => {
  initResizeHandles();
});
